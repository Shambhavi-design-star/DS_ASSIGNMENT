#include<iostream>
using namespace std;
/*write a game development program that implements the Bubble Sort algorithm. The program will simulate a simple game where the player can input a set of numbers, and the numbers will be sorted using Bubble Sort to simulate a "level-up" scenario where the player's scores are sorted in descending order.*/

int main()
{
	int i,n;
	cout<<"Enter number of rounds: ";
	cin>>n;
	int player1[n];
	int player2[n];
	
	cout<<"PLAYER 1 SCORES ARE:\n ";
	for(i=0;i<n;i++)
	{
		cout<<"round "<<i+1<<" score: ";
		cin>>player1[i];
	}
	cout<<"PLAYER 2 SCORES ARE:\n ";
	for(i=0;i<n;i++)
	{
		cout<<"round "<<i+1<<" score: ";
		cin>>player2[i];
	}
	int p,s,temp=0;
	for(p=0;p<n;p++)
	{
	    for(s=0;s<n-p-1;s++)
	    {
	        if(player1[s]<player1[s+1])
	        {
	            temp=player1[s];
	            player1[s]=player1[s+1];
	            player1[s+1]=temp;
	            }
	    }
	}
	cout<<"After sorting the scores of player 1 is:";
	for(i=0;i<n;i++)
	{
	    cout<<"\t"<<player1[i];
	}
	for(p=0;p<n;p++)
	{
	    for(s=0;s<n-p-1;s++)
	    {
	        if(player2[s]<player2[s+1])
	        {
	            temp=player2[s];
	            player2[s]=player2[s+1];
	            player2[s+1]=temp;
	            }
	    }
	}
	cout<<"\n After sorting the scores of player 2 is:";
	for(i=0;i<n;i++)
	{
	    cout<<"\t"<<player2[i];
	}
	if (player1[0]>player2[0])
	{
	 cout<<"\n ----PLAYER 1 IS THE WINNER----";
	}
	else if(player2[0]>player1[0])
	{
	    cout<<"\n---- PLAYER 2 IS THE WINNER----";
	}
	else
	{
	    cout<<"\n ----IT'S A TIE ^^----";
	}
return 0;
}
