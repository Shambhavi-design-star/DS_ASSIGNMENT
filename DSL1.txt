#include <iostream>
#include <iomanip> // For std::fixed and std::setprecision
/* Write a program for Tracking daily temperatures of 3 cities for a week . The program calculates the average temperature for each day and for the week.*/

using namespace std;

int main() {
    cout<<"Name:Shambhavi Vaishampayan\n PRN:B24CE1126\n";
    float temp[3][7];
    float avg = 0;
    float total = 0;
    int c, r;

    for (c = 0; c < 3; c++) {
        cout << "Enter temperature for city " << (c + 1) << " (7 days):\n";
        for (r = 0; r < 7; r++) {
            cout << "Day " << (r + 1) << ": ";
            cin >> temp[c][r];
        }
    }

    cout << "\ntemperature data:\n";
    printf("srno.\tCity\tD1\tD2\tD3\tD4\tD5\tD6\tD7\tAvg\n");
    for (c = 0; c < 3; c++) {
        cout << c + 1 << "\t";
        cout << "City " << (c + 1) << ":\t";
        total = 0;
        for (r = 0; r < 7; r++) {
            cout << temp[c][r] << "\t";
            total += temp[c][r];
        }
        avg = total / 7;
        cout << fixed << setprecision(2) << avg << "\n";
        cout << "----------------------------------------------------------------------" << endl;
    }

    // New section for daily averages across cities
    cout << "\nDaily Averages (across all cities):\n";
    cout << "Day\tAverage Temp\n";
    for (r = 0; r < 7; r++) {
        float daySum = 0;
        for (c = 0; c < 3; c++) {
            daySum += temp[c][r];
        }
        float dayAverage = daySum / 3.0; // Assuming 3 cities
        cout << "Day " << (r + 1) << "\t" << fixed << setprecision(2) << dayAverage << "\n";
    }

    return 0;
}
